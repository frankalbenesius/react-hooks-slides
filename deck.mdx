```jsx
import React, { useState } from "react";

function Counter({ initialCount }) {
  const [count, setCount] = useState(initialCount);
  return (
    <div>
      Count: {count}
      <button onClick={() => setCount(initialCount)}>Reset</button>
      <button onClick={() => setCount(prevCount => prevCount - 1)}>-</button>
      <button onClick={() => setCount(prevCount => prevCount + 1)}>+</button>
    </div>
  );
}
```

# useEffect

A place to put side effects like mutations, fetching, subscriptions, timers, and logging.

Runs after every component render. Replacing class lifecycle methods.

```
useEffect(() => {
  // do this thing after render
})
```

# useEffect - Hooks Example

```jsx
import React, {useEffect} from "react";

function Page({loggedIn, children}) {

  useEffect(() => {
    if (loggedIn) {
      document.title = "Online";
    }
  }, [loggedIn])

  return (
    <div>
      {children}
    </div>
  );
}
```

---

# useEffect - Fetch Example

```jsx
import React, {useEffect} from "react";

function CharacterList() {
  
  const [characters, setCharacters] = useState([]);
  
  useEffect(() => {
    fetch("url").then(data =>
      setCharacters(data);
    )}
  })

  return (
    <div>
      {characters.map(character => (
        <div>{character.name}</div>
      ))}
    </div>
  );
}
```
